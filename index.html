  -- Assuming table ACCOUNTS(acc_no, balance) exists
  PROCEDURE deposit(p_acc NUMBER, p_amt NUMBER) IS
  BEGIN UPDATE ACCOUNTS SET balance = balance + p_amt WHERE acc_no = p_acc; COMMIT; END;

  PROCEDURE withdraw(p_acc NUMBER, p_amt NUMBER) IS
  BEGIN UPDATE ACCOUNTS SET balance = balance - p_amt WHERE acc_no = p_acc; COMMIT; END;

  FUNCTION check_bal(p_acc NUMBER) RETURN NUMBER IS
    v_bal NUMBER;
  BEGIN SELECT balance INTO v_bal FROM ACCOUNTS WHERE acc_no = p_acc; RETURN v_bal; END;
END pkg_bank;
